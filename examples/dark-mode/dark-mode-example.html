Complete Dark Mode Implementation

JavaScript Implementation

function darkModeApp() {
    return {
        // State
        darkMode: false,
        systemPreference: 'unknown',
        supportsDarkMode: false,
        
        // Initialization
        init() {
            this.detectSystemPreference();
            this.loadSavedPreference();
            this.updateDarkMode();
        },
        
        // Toggle dark mode
        toggleDarkMode() {
            this.darkMode = !this.darkMode;
            this.updateDarkMode();
            this.savePreference();
        },
        
        // Apply dark mode to DOM
        updateDarkMode() {
            if (this.darkMode) {
                document.documentElement.classList.add('dark');
            } else {
                document.documentElement.classList.remove('dark');
            }
        },
        
        // Detect system preference
        detectSystemPreference() {
            if (window.matchMedia) {
                const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)');
                this.supportsDarkMode = true;
                this.systemPreference = mediaQuery.matches ? 'dark' : 'light';
                
                // Listen for changes
                mediaQuery.addEventListener('change', (e) => {
                    this.systemPreference = e.matches ? 'dark' : 'light';
                });
            }
        },
        
        // Save to localStorage
        savePreference() {
            localStorage.setItem('darkMode', this.darkMode.toString());
        },
        
        // Load from localStorage
        loadSavedPreference() {
            const saved = localStorage.getItem('darkMode');
            if (saved !== null) {
                this.darkMode = saved === 'true';
            } else {
                // Use system preference as default
                this.darkMode = this.systemPreference === 'dark';
            }
        }
    }
}

HTML Structure

<div x-data="darkModeApp()" 
     x-cloak 
     :class="darkMode ? 'dark' : ''" 
     class="min-h-screen bg-gray-50 dark:bg-gray-900">
     
  <button @click="toggleDarkMode()"
          :aria-label="darkMode ? 'Switch to light mode' : 'Switch to dark mode'"
          class="p-3 rounded-lg bg-gray-100 dark:bg-gray-700 hover:bg-gray-200 dark:hover:bg-gray-600">
    <i x-show="!darkMode" class="bi bi-moon"></i>
    <i x-show="darkMode" class="bi bi-sun"></i>
  </button>
  
  <!-- Your content with dark mode classes -->
  <div class="bg-white dark:bg-gray-800 text-gray-900 dark:text-gray-100">
    Content that adapts to theme
  </div>
</div>

                border-color 0.3s ease,

/* Prevent flash of unstyled content */
[x-cloak] { 
    display: none !important; 
}
